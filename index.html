                function wrap(value) {
                    return function () {
                        return value;
                    }
                }

                function bind(monad, f) {
                    return f(monad());
                }

                bind(monad, (value) => {
                    console.info(value);
                    return wrap(value);
                });
                monad = bind(monad, (value) => {
                    return wrap({ type: value });
                });
